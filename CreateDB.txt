CREATE DATABASE EntranceTestDB;
USE EntranceTestDB;
DROP TABLE IF EXISTS Tasks;
DROP TABLE IF EXISTS Users;

CREATE TABLE Users(
	id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	username VARCHAR(30) NOT NULL,
    pwd VARCHAR(30) NOT NULL
);

CREATE TABLE Tasks(
	id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	taskName VARCHAR(30) NOT NULL,
	taskDescription VARCHAR(30) NOT NULL,
	userId INT NOT NULL,
	expireDate VARCHAR(50) NOT NULL,
    taskStatus VARCHAR(10) NOT NULL,
    createdTime VARCHAR(50) NOT NULL,
    updatedTime VARCHAR(50) NOT NULL,
    FOREIGN KEY (userId)
        REFERENCES Users(id)
        ON DELETE CASCADE
);
